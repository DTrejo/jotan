### how to connect to a server

```js
var jotan = require('jotan')

var j = jotan(PORT, HOST)
j.send(new Buffer("c"))

setTimeout(function() {
  j.send({ life: 'rocks!' })

  // fails due to default 1000ms timeout on the server
  setTimeout(function() {
    j.send({ life: 'rocks!' })
    j.end()
  }, 1100)

}, 500)
```
